  SELECT FAS.FUND_INVESTMENT.FUND_NO,   
         FAS.FUND_INVESTMENT.TX_DATE,   
         FAS.FUND_INVESTMENT.PCT_DERIVATIVE,   
         FAS.FUND_INVESTMENT.PCT_DOMESTIC,   
         FAS.FUND_INVESTMENT.PCT_CHINA,   
         FAS.FUND_INVESTMENT.PCT_H_SHARES_RED_CHIPS,   
         FAS.FUND_INVESTMENT.FUM,   
         FAS.FUND.ORDINAL,   
         FAS.FUND_INVESTMENT.FUM_GLOBAL,   
         FAS.FUND_INVESTMENT.LONG_POSITION,   
         FAS.FUND_INVESTMENT.SHORT_POSITION,   
         FAS.FUND_INVESTMENT.ISSUE_SHARES,   
         FAS.FUND_INVESTMENT.ORG_CURRENCY_NO,   
         FAS.FUND_INVESTMENT.FUM_TOTAL,   
         FAS.FUND.CURRENCY_NO,
         FAS.FUND.HSBC_CODE,
         FAS.FUND.ISIN_CODE,
         FAS.FUND.FUND_MASTER_NO  ,
		FAS.FUND.SUBTA_NO
    FROM FAS.FUND_INVESTMENT,   
         FAS.FUND  
   WHERE ( FAS.FUND_INVESTMENT.FUND_NO = FAS.FUND.FUND_NO )   
ORDER BY FAS.FUND.ORDINAL ASC
